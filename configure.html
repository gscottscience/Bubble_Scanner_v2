// --- Test Scan logic --- const testScanBtn = document.getElementById('test-scan-btn'); const testScanUpload = document.getElementById('test-scan-upload'); const testScanStatus = document.getElementById('test-scan-status'); const testScanArea = document.getElementById('test-scan-area'); const testScanImg = document.getElementById('test-scan-img'); const testScanCanvas = document.getElementById('test-scan-canvas'); const testScanAnswers = document.getElementById('test-scan-answers'); testScanBtn.addEventListener('click', async () => { const file = testScanUpload.files[0]; if (!file) { alert('Please upload a filled scan image.'); return; } testScanStatus.textContent = 'Scanning...'; testScanArea.style.display = 'none'; testScanAnswers.innerHTML = ''; try { const formData = new FormData(); formData.append('bubble_sheet_image', file); // Use current config as test_name (simulate a test config upload) // We'll POST config as a string, backend must support this (or use a temp test name) // For now, just use the selected test name const selectedTest = testSelect.value; if (!selectedTest) throw new Error('Select a template first.'); formData.append('test_name', selectedTest); formData.append('grade_sheets', 'on'); const response = await fetch('/api/scan', { method: 'POST', body: formData }); const data = await response.json(); if (!data.results || !data.results[0]) throw new Error(data.error || 'No scan result.'); const result = data.results[0]; // Show scan image if (result.cropped_image_url) { testScanImg.src = '/' + result.cropped_image_url + `?t=${Date.now()}`; testScanImg.onload = () => drawTestScanOverlays(result); } // Show answers let html = 'Student ID: ' + (result.student_id || '(not detected)') + '
Answers:
'; (result.answers || []).forEach(a => { html += ``; }); html += '
Q#Answer
${a.q}${a.value || ''}
'; testScanAnswers.innerHTML = html; testScanArea.style.display = 'block'; testScanStatus.textContent = 'Scan complete!'; } catch (err) { testScanStatus.textContent = 'Error: ' + err.message; } }); function drawTestScanOverlays(result) { const img = testScanImg; if (!img.complete || !img.naturalWidth) return; testScanCanvas.width = img.naturalWidth; testScanCanvas.height = img.naturalHeight; testScanCanvas.style.width = img.width + 'px'; testScanCanvas.style.height = img.height + 'px'; const ctx = testScanCanvas.getContext('2d'); ctx.clearRect(0,0,testScanCanvas.width,testScanCanvas.height); // Use backend ROIs if available if (result.student_id_grid_roi) { drawRoi(ctx, result.student_id_grid_roi, '#6f42c1', 'Student ID ROI'); } if (result.answers_grid_col1_roi) { drawRoi(ctx, result.answers_grid_col1_roi, '#28a745', 'Answers Col 1'); } if (result.answers_grid_col2_upper_roi) { drawRoi(ctx, result.answers_grid_col2_upper_roi, '#fd7e14', 'Answers Col 2 Upper'); } if (result.answers_grid_col2_lower_roi) { drawRoi(ctx, result.answers_grid_col2_lower_roi, '#fd7e14', 'Answers Col 2 Lower'); } } function drawRoi(ctx, rect, color, label) { ctx.save(); ctx.strokeStyle = color; ctx.lineWidth = 3; ctx.strokeRect(rect[0], rect[1], rect[2], rect[3]); ctx.font = 'bold 18px sans-serif'; ctx.fillStyle = color; ctx.fillText(label, rect[0]+4, rect[1]+24); ctx.restore(); }
